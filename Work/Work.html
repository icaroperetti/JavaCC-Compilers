NON-TERMINALS

Start ::= ( COMANDOS ) <EOF>
DECLARAR_VARIAVEL ::= ( <TIPO> <LITERALS> )
ATRIBUICAO_VARIAVEL ::= <VALUES>
| <NUMEROS>
| <LITERALS>
| <BOOL>
IDENTIFICADOR ::= ( DECLARAR_VARIAVEL ( <VIRGULA> ( <LITERALS> | <VIRGULA> <LITERALS> )+ | <ATRIBUICAO> ATRIBUICAO_VARIAVEL )* <EOL> )
COMANDOS ::= ( WHILE | EXP_ARITMETICA | IDENTIFICADOR | IF | <COMMENT> | FUNCAO | DO_WHILE | SWITCH_CASE )+
CONDICAO ::= ( <ABRE_PARENTESES> EXP_BOOL <FECHA_PARENTESES> <ABRE_CHAVES> COMANDOS <FECHA_CHAVES> )*
IF ::= <IF> CONDICAO ( <ELSEIF> CONDICAO )* ( <ELSE> <ABRE_CHAVES> COMANDOS <FECHA_CHAVES> )*
WHILE ::= <WHILE> <ABRE_PARENTESES> EXP_BOOL <FECHA_PARENTESES> <ABRE_CHAVES> ( COMANDOS )* <FECHA_CHAVES>
FUNCAO ::= <FUNCTION> <ABRE_PARENTESES> ( PARAMETROS )* <FECHA_PARENTESES> <ABRE_CHAVES> ( COMANDOS )* <FECHA_CHAVES>
PARAMETROS ::= DECLARAR_VARIAVEL ( <VIRGULA> DECLARAR_VARIAVEL )*
EXP_BOOL ::= ( ( <VALUES> | <NUMEROS> | <LITERALS> ) <OPER_RELACIONAL> ( <VALUES> | <NUMEROS> | <LITERALS> ) )
EXP_ARITMETICA ::= ( <LITERALS> ( <INCREMENTO> | <ATRIBUICAO> ( <VALUES> | <NUMEROS> | <LITERALS> ) ) ( <OPER_ARITMETICO> ( <VALUES> | <NUMEROS> | <LITERALS> ) )* <EOL> )
ABRE_ESCOPO ::= <ABRE_CHAVES> ( COMANDOS ) <FECHA_CHAVES>
DO_WHILE ::= <DO> ABRE_ESCOPO <WHILE> <ABRE_PARENTESES> EXP_BOOL <FECHA_PARENTESES>
CASE ::= <CASE> ( <VALUES> | <LITERALS> ) <DOIS_PONTOS> COMANDOS <BREAK> <EOL>
SWITCH_CASE ::= <SWITCH> <ABRE_PARENTESES> <LITERALS> <FECHA_PARENTESES> <ABRE_CHAVES> ( CASE )* <FECHA_CHAVES>
